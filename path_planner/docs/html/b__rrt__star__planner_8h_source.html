488 template <class Map, class BoundingVolume, class Coordinates, class Path>
489 typename BRRT<Map, BoundingVolume, Coordinates, Path>::Node *
490 BRRT<Map, BoundingVolume, Coordinates, Path>::nearestVertex(
491  Coordinates *xRand, std::vector<std::unique_ptr<Node>> *tree) {
492  double previousDistance{std::numeric_limits<double>::max()};
493  double currentDistance{0.0};
494  Node *nearest = nullptr;
495  // We could use the in-built Point difference and norm method as well.
496  auto norm = [&](const auto &A, const auto &B) mutable {
497  currentDistance = std::sqrt(((A.x() - B.x()) * (A.x() - B.x())) +
498  ((A.y() - B.y()) * (A.y() - B.y())) +
499  ((A.z() - B.z()) * (A.z() - B.z())));
500  };
501  for (auto &vertex : *tree) {
502  // Compute the euclidean distance to the random point
503  norm(*xRand, vertex->point);
504  if (currentDistance < previousDistance) {
505  previousDistance = currentDistance;
506  nearest = vertex.get();
507  }
508  }
509  return nearest;
510 }

758 template <class Map, class BoundingVolume, class Coordinates, class Path>
759 Path *BRRT<Map, BoundingVolume, Coordinates, Path>::computePath(
760  Coordinates const &start_, Coordinates const &goal_) {
761  // Main B-RRTStar algorithm body
762  try {
763  if (isInCollision(start_))
764  throw std::runtime_error(
765  "Start coordinates are in Occupied or Unknown space!");
766  if (isInCollision(goal_))
767  throw std::runtime_error(
768  "Goal coordinates are in Occupied or Unknown space!");
769  } catch (const std::exception &e) {
770  std::cout << "Cannot compute path, reason: " << e.what() << std::endl;
771  return nullptr;
772  }
773  std::cout << "Path planning started" << std::endl;
774  // Initialize nodes
775  Node *xNearest = nullptr;
776  Node *xConn = nullptr;
777  Node *xMin = nullptr;
778  // Useful if running the globalPlanner more than once.
779  treeA.clear();
780  treeB.clear();
781  path.clear();
782  treeAPtr = &treeA;
783  treeBPtr = &treeB;
784  //
785  auto xRand = std::make_unique<Coordinates>();
786  auto xNew = std::make_unique<Coordinates>();
787  auto startPoint = std::make_unique<Node>(start_);
788  auto goalPoint = std::make_unique<Node>(goal_);
789  //
790  treeA.emplace_back(std::move(startPoint)); // Add start node.
791  treeB.emplace_back(std::move(goalPoint)); // Add goal node.
792  //
793  sigmaF = std::numeric_limits<double>::max();
794  auto start = steady_clock::now();
795  auto elapsedTime = std::numeric_limits<long>::min();
796  int iter = iterations;
797  // Main Loop
798  while (iter--) {
799  // Sample
800  *xRand = sample();
801  // Get nearest vertex
802  xNearest = nearestVertex(xRand.get(), treeAPtr);
803  *xNew = extend(&(xNearest->point), xRand.get());
804  // Get the nearby vertices.
805  nearVertices(xNew.get(), treeAPtr);
806  if (nearby.empty())
807  nearby.emplace_back(xNearest);
808  // Sort nearby vertices by cost and select the best parent option.
809  xMin = chooseBestParent(xNew.get(), treeAPtr);
810  if (xMin != nullptr) {
811  expandTree(xMin, xNew.get(), treeAPtr);
812  xConn = nearestVertex(xNew.get(), treeBPtr);
813  connect(xNew.get(), &(xConn->point), treeBPtr);
814  }
815  // Swap Trees
816  std::swap(treeAPtr, treeBPtr);
817  elapsedTime = duration_cast<seconds>(steady_clock::now() - start).count();
818  }
819  return (path.empty()) ? nullptr : &path;
820 }
