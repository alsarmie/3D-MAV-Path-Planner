458 template <class Map, class BoundingVolume, class Coordinates, class Path>
459 bool PIBRRT<Map, BoundingVolume, Coordinates, Path>::isInCollision(
460  Coordinates const &goal_, Coordinates const &position_) {
461  Coordinates direction = goal_ - position_;
462  Coordinates center = position_ + (direction / 2.0);
463  double distance = direction.norm();
464  direction /= distance; // Distance normalization
465  // Calculate yaw, pitch and roll for oriented bounding box
466  double yaw = -atan2(direction[1], direction[0]);
467  double pitch = -asin(direction[2]);
468  double roll = 0;
469  // Create an oriented bounding box between position and goal, with a size
470  // radius.
471  std::lock_guard lck(mapMtx);
472  mapDone = false;
473  ufo::geometry::OBB obb(center,
474  ufo::math::Vector3(distance / 2.0, radius, radius),
475  ufo::math::Quaternion(roll, pitch, yaw));
476  // Iterate through all leaf nodes that intersects the bounding volume
477  // Check intersection with //occupied/free/unknown space
478  for (auto it = map->beginLeaves(obb, true, false, true, false, 0),
479  it_end = map->endLeaves();
480  it != it_end; ++it) {
481  // Is in collision since a leaf node intersects the bounding volume.
482  mapDone = true;
483  mapCV.notify_one();
484  return true;
485  }
486  mapDone = true;
487  mapCV.notify_one();
488  // No leaf node intersects the bounding volume.
489  return false;
490 }

714  bool isTreeA) {
715  // Get sorted List
716  double cost{0.0};
717  std::pair<double, Node *> n;
718  {
719  std::lock_guard lck(getSortedListMtx);
720  (isTreeA) ? (getSortedListADone = false) : (getSortedListBDone = false);
721  }
722  {
723  std::lock_guard lckA((isTreeA) ? treeAMtx : treeBMtx);
724  std::lock_guard lckB((isTreeA) ? xNearAMtx : xNearBMtx);
725  std::lock_guard lckC((isTreeA) ? listAMtx : listBMtx);
726  auto tree = (isTreeA) ? treeAPtr : treeBPtr;
727  auto xNear = (isTreeA) ? xNearA : xNearB;
728  for (auto &vertex : xNear) {
729  cost =
730  norm((*tree)[0]->point, vertex->point) + norm(vertex->point, xRand);
731  n.first = cost;
732  n.second = vertex;
733  (isTreeA) ? listA.emplace_back(std::move(n))
734  : listB.emplace_back(std::move(n));
735  }
736  }
737  {
738  std::lock_guard lckC((isTreeA) ? listAMtx : listBMtx);
739  std::sort((isTreeA) ? listA.begin() : listB.begin(),
740  (isTreeA) ? listA.end() : listB.end(),
741  [](const auto &a, const auto &b) { return a.first < b.first; });
742  }
743  std::lock_guard lck(getSortedListMtx);
744  (isTreeA) ? (getSortedListADone = true) : (getSortedListBDone = true);
745  (isTreeA) ? getSortedListACV.notify_one() : getSortedListBCV.notify_one();

861  this, *xRand, false));
862  {
863  std::unique_lock lck(getBestTreeParentMtx);
864  getBestTreeParentACV.wait(lck,
865  [this] { return getBestTreeParentADone.load(); });
866  getBestTreeParentBCV.wait(lck,
867  [this] { return getBestTreeParentBDone.load(); });
868  }
869 
870  if (xMinA != nullptr && xMinB != nullptr) {
871 
872  if (minCostA <= minCostB) {
873  xMin = xMinA;
874  } else if (minCostB < minCostA) {
875  xMin = xMinB;
876  insertInA = false;
877  }
878  if (connection)
879  connectTrees(xRand);
880  } else if (xMinA != nullptr) {
881 
882  xMin = xMinA;
883  } else if (xMinB != nullptr) {
884 
885  xMin = xMinB;
886  insertInA = false;
887  }

904 template <class Map, class BoundingVolume, class Coordinates, class Path>
905 void PIBRRT<Map, BoundingVolume, Coordinates, Path>::rewire(
906  std::vector<std::unique_ptr<Node>> *tree) {
907  double cost{0.0};
908  auto xRand = tree->back().get();
909  auto xInit = (*tree)[0].get();
910  auto list = (insertInA.load()) ? listA : listB;
911  for (auto &vertex : list) {
912  cost = norm(xInit->point, xRand->point) +
913  norm(xRand->point, vertex.second->point);
914  {
915  std::unique_lock lckMap(mapMtx);
916  mapCV.wait(lckMap, [this] { return mapDone.load(); });
917  }
918  if (cost < norm(xInit->point, vertex.second->point))
919  if (!isInCollision(xRand->point, vertex.second->point)) {
920  vertex.second->parent = xRand;
921  vertex.second->costToParent = norm(xRand->point, vertex.second->point);
922  }
923  }
924 }

973 template <typename Map, typename BoundingVolume, typename Coordinates,
974  typename Path>
975 void PIBRRT<Map, BoundingVolume, Coordinates, Path>::connectTrees(
976  Coordinates *xRand) {
977  double costA{0.0};
978  double costB{0.0};
979  double sigmaAB{0.0};
980  Path pathA;
981  Path pathB;
982  {
983  std::unique_lock lckMap(mapMtx);
984  mapCV.wait(lckMap, [this] { return mapDone.load(); });
985  }
986  if (!isInCollision(xMinA->point, *xRand) &&
987  !isInCollision(xMinB->point, *xRand)) {
988  pathA = traceBack(xMinA, costA);
989  pathB = traceBack(xMinB, costB);
990  std::reverse(pathB.begin(), pathB.end());
991  pathA.emplace_back(*xRand);
992  sigmaAB =
993  costA + costB + norm(xMinA->point, *xRand) + norm(xMinB->point, *xRand);
994  if (sigmaAB < sigmaF) { // Store new path.
995  path.clear();
996  sigmaF = sigmaAB;
997  path.resize(pathA.size() + pathB.size());
998  std::move(pathA.begin(), pathA.end(), path.begin());
999  std::move(pathB.begin(), pathB.end(), path.begin() + pathA.size());
1000  }
1001  }
1002 }
1018 template <class Map, class BoundingVolume, class Coordinates, class Path>
1019 Path PIBRRT<Map, BoundingVolume, Coordinates, Path>::traceBack(Node *node,
1020  double &cost) {
1021  Path p;
1022  // Get goal
1023  while (node != nullptr) { // get the path in one vector
1024  p.emplace_front(node->point);
1025  cost += node->costToParent;
1026  node = node->parent;
1027  }
1028  return p;
1029 }
1043 template <class Map, class BoundingVolume, class Coordinates, class Path>
1044 Path *PIBRRT<Map, BoundingVolume, Coordinates, Path>::computePath(
1045  Coordinates const &start_, Coordinates const &goal_) {

1047  try {
1048  if (isInCollision(start_))
1049  throw std::runtime_error(
1050  "Start coordinates are in Occupied or Unknown space!");
1051  if (isInCollision(goal_))
1052  throw std::runtime_error(
1053  "Goal coordinates are in Occupied or Unknown space!");
1054  } catch (const std::exception &e) {
1055  std::cout << "Cannot compute path, reason: " << e.what() << std::endl;
1056  return nullptr;
1057  }
1058  std::cout << "Path planning started" << std::endl;
1059  // Initialize nodes
1060  Node *xMin = nullptr;
1061  // Useful if running the globalPlanner more than once.
1062  treeA.clear();
1063  treeB.clear();
1064  path.clear();
1065  treeAPtr = &treeA;
1066  treeBPtr = &treeB;
1067  //
1068  auto xRand = std::make_unique<Coordinates>();
1069  auto startPoint = std::make_unique<Node>(start_);
1070  auto goalPoint = std::make_unique<Node>(goal_);
1071  //
1072  treeA.emplace_back(std::move(startPoint)); // Add start node.
1073  treeB.emplace_back(std::move(goalPoint)); // Add goal node.
1074  //
1075  sigmaF = std::numeric_limits<double>::max();
1076  int iter = iterations;
1077  // Main Loop
1078  while (iter--) { //
1079  connection = true;
1080  // Sample
1081  *xRand = sample();
1082  // Get the nearby vertices for both trees.
1083  {
1084  nearVertices(xRand.get());
1085  std::unique_lock lck(nearVerticesMtx);
1086  nearVerticesACV.wait(lck, [this] { return nearVerticesADone.load(); });
1087  nearVerticesBCV.wait(lck, [this] { return nearVerticesBDone.load(); });
1088  }
1089  if (xNearA.empty() || xNearB.empty()) {
1090  nearestVertex(xRand.get());
1091  {
1092  std::unique_lock lck(nearestVertexMtx);
1093  nearestVertexACV.wait(lck,
1094  [this] { return nearestVertexADone.load(); });
1095  nearestVertexBCV.wait(lck,
1096  [this] { return nearestVertexBDone.load(); });
1097  }
1098  connection = false;
1099  }
1100  // Sort group of near vertices.
1101  getSortedList(xRand.get());
1102  {
1103  std::unique_lock lck(getSortedListMtx);
1104  getSortedListACV.wait(lck, [this] { return getSortedListADone.load(); });
1105  getSortedListBCV.wait(lck, [this] { return getSortedListBDone.load(); });
1106  }
1107  xMin = getBestTreeParent(xRand.get());
1108 
1109  if (xMin != nullptr)
1110  expandTree(xMin, xRand.get(), (insertInA.load()) ? treeAPtr : treeBPtr);
1111  }
1112 
1113  return (path.empty()) ? nullptr : &path;
1114 }
