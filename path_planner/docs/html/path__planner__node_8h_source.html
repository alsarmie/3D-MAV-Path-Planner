124 void pathPlanner::run() {
125  // Wait for map before trying to compute a path!
126  while (!mapAvailable)
127  ros::spinOnce();
128  // Until shutdown is executed, retrieve 3D points from Rviz provided by user.
129  while (ros::ok()) {
130  path = nullptr;
131  getRvizPoints(); // Waiting function, returns when the pointCallback
132  // receives new points.
133  publishPointsMarkers();
134  // path = rrt_planner.computePath(points[0], points[1]);
135  std::visit(overload{[&](RRTS &planner) {
136  path = planner.computePath(points[0], points[1]);
137  },
138  [&](BRRT &planner) {
139  path = planner.computePath(points[0], points[1]);
140  },
141  [&](IBRRT &planner) {
142  path = planner.computePath(points[0], points[1]);
143  },
144  [&](PIBRRT &planner) {
145  path = planner.computePath(points[0], points[1]);
146  }},
147  rrt_planner);
148 
149  if (path != nullptr) {
150  interpolatedPath = catmullRomSpline.interpolate(10.0, path);
151  if (!interpolatedPath.empty()) {
152  ROS_INFO("Path globalPlanner found path!");
153  publishPath();
154  publishInterpolatedPath();
155  ROS_INFO("Path publish!");
156  } else {
157  ROS_ERROR("NO PATH GENERATED!");
158  }
159  } else {
160  ROS_ERROR("NO PATH GENERATED!");
161  }
162  }
163 }

220 void pathPlanner::publishInterpolatedPath() {
221  double yaw{0.0};
222  double pitch{0.0};
223  double roll{0.0};
224  nav_msgs::Path path_;
225  path_.header.frame_id = "map";
226  tf2::Quaternion q;
227 
228  // Convert to ROS nav_msgs/Path format.
229  for (auto &point : interpolatedPath) {
230  geometry_msgs::PoseStamped pose;
231  pose.header.frame_id = "map";
232  pose.pose.position.x = point.x();
233  pose.pose.position.y = point.y();
234  pose.pose.position.z = point.z();
235  point /= point.norm();
236  yaw = -atan2(point.y(), point.x());
237  pitch = -asin(point.y());
238  q.setRPY(roll, pitch, yaw);
239  pose.pose.orientation.x = q.getX();
240  pose.pose.orientation.y = q.getY();
241  pose.pose.orientation.z = q.getZ();
242  pose.pose.orientation.w = q.getW();
243  path_.poses.emplace_back(pose);
244  }
245  while (pathIntPublisher.getNumSubscribers() < 1)
246  ros::spinOnce(); // sleep(1);
247  path_.header.stamp = ros::Time::now();
248  pathIntPublisher.publish(path_);
249  interpolatedPath.clear();

322 void pathPlanner::setPlannerType() {
323  switch (type) {
324  case 0: // RRT*
325  rrt_planner =
326  RRTS(&map, min_, max_, delta_, eps_, thresh_, searchRadius_, radius_);
327  ROS_INFO("RRT* path globalPlanner ready!");
328  break;
329  case 1: // B-RRT*
330  rrt_planner =
331  BRRT(&map, min_, max_, delta_, searchRadius_, radius_, iterations_);
332  ROS_INFO("B-RRT* path globalPlanner ready!");
333  break;
334  case 2: // IB-RRT*
335  rrt_planner =
336  IBRRT(&map, min_, max_, delta_, searchRadius_, radius_, iterations_);
337  ROS_INFO("IB-RRT* path globalPlanner ready!");
338  break;
339  case 3:
340  rrt_planner = PIBRRT(&map, min_, max_, delta_, searchRadius_, radius_,
341  iterations_, workers_);
342  ROS_INFO("Parallel IB-RRT* path globalPlanner ready!");
343  break;
344  default:
345  ROS_WARN("No planner selected! Using RRT* as default option!");
346  rrt_planner =
347  RRTS(&map, min_, max_, delta_, eps_, thresh_, searchRadius_, radius_);
348  ROS_INFO("RRT* path globalPlanner ready!");
349  }
