398 bool IBRRT<Map, BoundingVolume, Coordinates, Path>::isInCollision(
399  Coordinates const &goal_, Coordinates const &position_) {
400  Coordinates direction = goal_ - position_;
401  Coordinates center = position_ + (direction / 2.0);
402  double distance = direction.norm();
403  direction /= distance; // Distance normalization
404  // Calculate yaw, pitch and roll for oriented bounding box
405  double yaw = -atan2(direction[1], direction[0]);
406  double pitch = -asin(direction[2]);

412  ufo::math::Quaternion(roll, pitch, yaw));

489 void IBRRT<Map, BoundingVolume, Coordinates, Path>::nearestVertex(
490  Coordinates *xRand) {
491  // Finding the nearest node to the random point in each tree.
492  xNearA.clear();
493  xNearB.clear();
494  auto nearest = [&](const auto *tree, auto &xNear) mutable {
495  Node *nearest{nullptr};
496  double previousDistance{std::numeric_limits<double>::max()};
497  double currentDistance{0.0};
498  for (auto &vertex : *tree) {
499  // Compute the euclidean distance to the random point
500  currentDistance = norm(*xRand, vertex->point);
501  if (currentDistance < previousDistance) {
502  previousDistance = currentDistance;
503  nearest = vertex.get();
504  }
505  }
506  xNear.emplace_back(std::move(nearest));
507  };
508 
509  nearest(treeAPtr, xNearA);
510  nearest(treeBPtr, xNearB);

526 template <class Map, class BoundingVolume, class Coordinates, class Path>
527 void IBRRT<Map, BoundingVolume, Coordinates, Path>::nearVertices(
528  Coordinates *xRand) {
529  // Finding the near vertices to the random point within a ball or radius R.
530  xNearA.clear();
531  xNearB.clear();
532  auto limit = [&](const auto &size) {
533  return searchRadius *
534  std::pow(std::log((double)size) / (double)size, 1.0 / 3.0);
535  };
536  auto near = [&](const auto *tree, auto &xNear) mutable {
537  for (auto &vertex : *tree)
538  // If the point is within the search sphere centered at xRand
539  if (norm(vertex->point, *xRand) <= limit(tree->size()))
540  xNear.emplace_back(vertex.get());
541  };
542  near(treeAPtr, xNearA);
543  near(treeBPtr, xNearB);

601 template <class Map, class BoundingVolume, class Coordinates, class Path>
602 typename IBRRT<Map, BoundingVolume, Coordinates, Path>::Node *
603 IBRRT<Map, BoundingVolume, Coordinates, Path>::getBestTreeParent(
604  bool connection, Coordinates *xRand) {
605  insertInA = true;
606  double minCostA{0.0};
607  double minCostB{0.0};
608  Node *xMin = nullptr;
609  auto getBestParent = [&](const auto &list, auto &cost) mutable {
610  // Choose best parent
611  for (auto &vertex : list)
612  if (!isInCollision(vertex.second->point, *xRand)) {
613  cost = vertex.first;
614  return vertex.second;
615  }
616  return (Node *)nullptr;
617  };
618  auto xMinA = getBestParent(listA, minCostA);
619  auto xMinB = getBestParent(listB, minCostB);
620  if (xMinA != nullptr && xMinB != nullptr) {
621  if (minCostA <= minCostB) {
622  xMin = xMinA;
623  } else if (minCostB < minCostA) {
624  xMin = xMinB;
625  insertInA = false;
626  }
627  if (connection)
628  connectTrees(xMinA, xMinB, xRand);
629  } else if (xMinA != nullptr) {
630  xMin = xMinA;
631  } else if (xMinB != nullptr) {
632  xMin = xMinB;
633  insertInA = false;
634  }
635  return xMin;
636 }

652 void IBRRT<Map, BoundingVolume, Coordinates, Path>::rewire(
653  std::vector<std::unique_ptr<Node>> *tree) {
654  double cost{0.0};
655  auto xRand = tree->back().get();
656  auto xInit = (*tree)[0].get();

662  vertex.second->parent = xRand;

664  }

781 template <class Map, class BoundingVolume, class Coordinates, class Path>
782 Path *IBRRT<Map, BoundingVolume, Coordinates, Path>::computePath(
783  Coordinates const &start_, Coordinates const &goal_) {
784  // Main IB-RRTStar algorithm body
785  try {
786  if (isInCollision(start_))
787  throw std::runtime_error(
788  "Start coordinates are in Occupied or Unknown space!");
789  if (isInCollision(goal_))
790  throw std::runtime_error(
791  "Goal coordinates are in Occupied or Unknown space!");
792  } catch (const std::exception &e) {
793  std::cout << "Cannot compute path, reason: " << e.what() << std::endl;
794  return nullptr;
795  }
796  std::cout << "Path planning started" << std::endl;
797  // Initialize nodes
798  Node *xMin = nullptr;
799  // Useful if running the globalPlanner more than once.
800  treeA.clear();
801  treeB.clear();
802  path.clear();
803  treeAPtr = &treeA;
804  treeBPtr = &treeB;
805  //
806  auto xRand = std::make_unique<Coordinates>();
807  auto startPoint = std::make_unique<Node>(start_);
808  auto goalPoint = std::make_unique<Node>(goal_);
809  //
810  treeA.emplace_back(std::move(startPoint)); // Add start node.
811  treeB.emplace_back(std::move(goalPoint)); // Add goal node.
812  //
813  sigmaF = std::numeric_limits<double>::max();
814  int iter = iterations;
815  // Main Loop
816  while (iter--) {
817  connection_ = true;
818  // Sample
819  *xRand = sample();
820  // Get the nearby vertices for both trees.
821  nearVertices(xRand.get());
822  if (xNearA.empty() || xNearB.empty()) {
823  nearestVertex(xRand.get());
824  connection_ = false;
825  }
826  // Sort group of near vertices.
827  getSortedList(xRand.get());
828  xMin = getBestTreeParent(connection_, xRand.get());
829  if (xMin != nullptr)
830  expandTree(xMin, xRand.get(), (insertInA) ? treeAPtr : treeBPtr);
831  }
832 
833  return (path.empty()) ? nullptr : &path;
834 }
