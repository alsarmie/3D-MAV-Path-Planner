467 RRTStar<Map, BoundingVolume, Coordinates, Path>::nearestVertex(Node *qrand) {

470  double currentDistance{0.0};

472  // We could use the in-built Point difference and norm method as well.
473  auto norm = [&](const auto &A, const auto &B) mutable {

475  ((A.y() - B.y()) * (A.y() - B.y())) +

505 template <class Map, class BoundingVolume, class Coordinates, class Path>
506 Coordinates RRTStar<Map, BoundingVolume, Coordinates, Path>::steer(
507  Coordinates const &nearestVertex, Coordinates const &randomVertex) {
508  // Assuming holonomic motion model.
509  Coordinates difference = randomVertex - nearestVertex;
510  double norm = difference.norm();
511  if (norm > deltaStep) {
512  // Random point is beyond our step size
513  difference /= norm; // unit vector;
514  return nearestVertex +
515  difference * deltaStep; // return a point at step distance in
516  // direction of the random point.
517  }
518  return randomVertex;
519 }
533 template <class Map, class BoundingVolume, class Coordinates, class Path>
534 void RRTStar<Map, BoundingVolume, Coordinates, Path>::chooseParent(
535  Node *qNear, Node *newVertex) {
536  double cost{0.0};
537  double limit =
538  searchRadius *
539  std::pow(std::log((double)tree.size()) / (double)tree.size(), 1.0 / 3.0);
540  nearby.clear();
541  auto norm = [](const auto &A, const auto &B) {
542  return std::sqrt(((A.x() - B.x()) * (A.x() - B.x())) +
543  ((A.y() - B.y()) * (A.y() - B.y())) +
544  ((A.z() - B.z()) * (A.z() - B.z())));
545  };
546  for (auto &vertex : tree)
547  if (vertex.get() != qNear && norm(vertex->point, newVertex->point) <= limit)
548  nearby.emplace_back(vertex.get());
549  for (auto &vertex : nearby)
550  if (!isInCollision(vertex->point, newVertex->point)) {
551  cost = norm(vertex->point, newVertex->point) + vertex->costToParent;
552  if (cost < newVertex->costToParent) {
553  newVertex->parent = vertex;
554  vertex->child = newVertex;
555  newVertex->costToParent = cost;
556  }
557  }
558 }

573 void RRTStar<Map, BoundingVolume, Coordinates, Path>::rewire() {
574  double cost{0.0};
575  auto norm = [](const auto &A, const auto &B) {
576  return std::sqrt(((A.x() - B.x()) * (A.x() - B.x())) +
577  ((A.y() - B.y()) * (A.y() - B.y())) +
578  ((A.z() - B.z()) * (A.z() - B.z())));
579  };
580  auto qNew = tree.back().get();
581  for (auto &vertex : nearby) {
582  cost = qNew->costToParent + norm(qNew->point, vertex->point);
583  if (cost < vertex->costToParent) {
584  qNew->child = vertex;
585  vertex->parent->child = nullptr;
586  vertex->parent = qNew;
587  vertex->costToParent = cost;
588  }
589  }

608 template <class Map, class BoundingVolume, class Coordinates, class Path>
609 void RRTStar<Map, BoundingVolume, Coordinates, Path>::expandTree(
610  Node *qNear, Coordinates qNew) {
611 
612  auto norm = [](const auto &A, const auto &B) {
613  return std::sqrt(((A.x() - B.x()) * (A.x() - B.x())) +
614  ((A.y() - B.y()) * (A.y() - B.y())) +
615  ((A.z() - B.z()) * (A.z() - B.z())));
616  };
617  std::unique_ptr<Node> newVertex = std::make_unique<Node>(std::move(qNew));
618  qNear->child = newVertex.get();
619  newVertex->parent = qNear;
620  newVertex->costToParent = norm(qNear->point, newVertex->point) +
621  qNear->costToParent; // cost so far.
622  // Look for the nearest neighbor vertices of the new point and choose the best
623  // parent based on cost.
624  chooseParent(qNear, newVertex.get());
625  tree.emplace_back(std::move(newVertex));
626  // Rewire the tree
627  rewire();
628 }

664 template <class Map, class BoundingVolume, class Coordinates, class Path>
665 Path *RRTStar<Map, BoundingVolume, Coordinates, Path>::computePath(
666  Coordinates const &start_, Coordinates const &goal_) {
667  // Main RRTStar algorithm body
668  try {
669  if (isInCollision(start_))
670  throw std::runtime_error(
671  "Start coordinates are in Occupied or Unknown space!");
672  if (isInCollision(goal_))
673  throw std::runtime_error(
674  "Goal coordinates are in Occupied or Unknown space!");
675  } catch (const std::exception &e) {
676  std::cout << "Cannot compute path, reason: " << e.what() << std::endl;
677  return nullptr;
678  }
679  std::cout << "RRT* Path planning started" << std::endl;
680 
681  tree.clear(); // Useful if running the globalPlanner more than once.
682  path.clear();
683 
684  double distanceToGoal = std::numeric_limits<double>::max();
685  // Initialize nodes
686  Coordinates qNew;
687  Node *qNearest = nullptr;
688 
689  auto qStart = std::make_unique<Node>(start_);
690  auto qRand = std::make_unique<Node>();
691  tree.emplace_back(std::move(qStart)); // Add start node.
692  auto start = steady_clock::now();
693  auto elapsedTime = std::numeric_limits<long>::min();
694  // Run algorithm until it reaches a timeout or finds a path with an error <=
695  // epsilon to the goal.
696  while ((elapsedTime < timeout) && distanceToGoal >= epsilon) {
697  if (rnd(randomEngine) > threshold)
698  qRand->point = goal_;
699  else
700  qRand->point = generateRandomPoint();
701  qNearest = nearestVertex(qRand.get());
702  qNew = steer(qNearest->point, qRand->point);
703  if (!isInCollision(qNearest->point, qNew)) {
704  // Add point to tree if it is collision free
705  expandTree(qNearest, qNew);
706  distanceToGoal = (goal_ - qNew).norm(); // Update the distance
707  // left.
708  }
709  elapsedTime = duration_cast<seconds>(steady_clock::now() - start).count();
710  }
711  // We need to connect the goal to the closest vertex if distanceToGoal <=
712  // epsilon and n>0
713  std::unique_ptr<Node> qGoal = std::make_unique<Node>(goal_);
714  qNearest = nearestVertex(qGoal.get());
715  if (!isInCollision(qNearest->point,
716  qGoal->point)) { // We can get to the goal directly
717  qNearest->child = qGoal.get();
718  qGoal->parent = qNearest;
719  tree.emplace_back(std::move(qGoal));
720  } else {
721  // We did not reach the goal, we return the path that was the closest.
722  for (auto it = tree.begin(); it != tree.end(); it++)
723  // move the closest vertex to the goal to the end of the vector,
724  // so it is easier to backtrack.
725  if (it->get() == qNearest)
726  std::rotate(it, it + 1, tree.end());
727  }
728  traceBack();
729  return (path.empty()) ? nullptr : &path;
730 }
